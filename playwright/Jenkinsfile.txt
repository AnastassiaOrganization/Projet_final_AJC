pipeline {
  agent any
  stages {
    stage('install playwright') {
      steps {
        bat 'npm i -D @playwright/test'
        bat 'npx playwright install'
      }
    }
    stage('test') {
      steps {
        sh '''
          npx playwright test --list
          npx playwright test
        '''
      }
      post {
        success {
          archiveArtifacts(artifacts: 'homepage-*.png', followSymlinks: false)
          sh 'rm -rf *.png'
        }
      }
    }
  }
}
